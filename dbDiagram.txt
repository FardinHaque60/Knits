// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table follows {
  following_user_id integer
  followed_user_id integer
  created_at timestamp 
}

Table users {
  id integer [primary key]
  username varchar
  password varchar
  role varchar
  email varchar
  phoneNumber varchar
  created_at timestamp
  calendarID [foreign key]
}
Table messages {
  messageID integer [primary key]
  userID integer 
  content text
  timestamp timestamp
}

Table cars {
  carID integer [primary key]
  driverID integer 
  passengers text
  seatsAvailable integer
}

Table posts {
  id integer [primary key]
  title varchar
  body text [note: 'Content of the post']
  user_id integer
  status varchar
  created_at timestamp
}

Table comments {
  commentID integer [primary key]
  postID integer
}

Table reviews {
  reviewID integer [primary key]
  userID integer [foreign key]
  locationID integer [foreign key]
}

Table locations {
  locationID integer [primary key]
  reviewID [foreign key]
  name varchar
  address varchar
}

Table calendar {
  calendarID integer [primary key]
  userID integer [foreign key]
  calendars varchar
  blocks varchar
}

Ref: posts.user_id > users.id // many-to-one

Ref: messages.userID > users.id

Ref: users.id > cars.carID

Ref: users.id < follows.following_user_id

Ref: comments.postID > posts.id

Ref: users.id < follows.followed_user_id

Ref: reviews.userID > users.ID

Ref: reviews.locationID > locations.locationID

Ref: locations.reviewID > reviews.reviewsID

Ref: users.calendarID > calendar.calendarID

Ref: calendar.userID > users.ID
