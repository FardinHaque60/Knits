Table users {
  Userid integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}

Table posts {
  Postid integer [primary key]
  title varchar
  body text [note: 'Content of the post']
  user_id integer
  timestamp timestamp
}

Table comments {
  commentID integer [primary key]
}

Ref: posts.user_id > users.Userid // many-to-one
